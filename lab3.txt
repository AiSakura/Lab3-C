int mass(int n1, int m1, int k1, int z1, int h1, int* a1)
{
    int t;

    for (int i = k1; i < m1 * z1 - h1; i = (m1 + 1) + i) {
        for (int j = k1; j < m1 * z1 - i - h1; j = j + (1 + m1)) {
            if (a1[j] > a1[j + 1 + m1])
            {

                t = a1[j]; a1[j] = a1[j + 1 + m1]; a1[j + 1 + m1] = t;

            }
        }
    }

    return *a1;
}

}
